\chapter*{Glossary of Research Terms}
\addcontentsline{toc}{chapter}{Glossary of Research Terms}

\begin{longtable}{p{2.7cm}p{10.2cm}}
    \toprule
    \textbf{Term} & \textbf{Definition} \\
    \midrule\endhead
    BAG & \textit{Basisregistratie Adressen en Gebouwen}: the Dutch key register of addresses and buildings. \\
    BRP & \textit{Basisregistratie Personen}: the municipal register of residents in the Netherlands. \\
    CBS & \textit{Centraal Bureau voor de Statistiek} (Statistics Netherlands), the national statistical office. \\
    DM & Decision Maker: a synthetic probability distribution obtained by pooling the experts' distributions. \\
    EWDM & Equal Weight Decision Maker: a pool that assigns the same weight to every expert. \\
    PWDM & Performance Weight Decision Maker: a pool weighted by experts' calibration and information scores. \\
    Primos & A demographic forecasting model used in the Netherlands for population, household, and housing-demand projections. \\
    Seed variable & A calibration question from the experts' domain whose realization is known to the analyst but hidden during elicitation. \\
    Target variable & A quantity of substantive interest whose realization is unknown at the time of elicitation. \\
    5th percentile ($q_{0.05}$) & A value for which the expert assigns a 5\% probability that the realization will be lower. \\
    Median ($q_{0.50}$) & A value that divides the expert's distribution into equally probable lower and upper halves. \\
    95th percentile ($q_{0.95}$) & A value for which the expert assigns a 5\% probability that the realization will be higher. \\
    WOZ & \textit{Waardering Onroerende Zaken}: the statutory assessed value of Dutch real estate used for taxation and administrative purposes. \\
    \bottomrule
\end{longtable}
